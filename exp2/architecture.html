4. 数据库物理模型（可执行 SQL 语句）

4.1 创建数据表 (CREATE TABLE)

-- ============================================
-- 学生管理系统 - SQLite 数据库建表脚本
-- ============================================

-- 1. 创建教师表
CREATE TABLE IF NOT EXISTS teachers (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name VARCHAR(50) NOT NULL,
    title VARCHAR(50),
    department VARCHAR(100) NOT NULL,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

-- 2. 创建学生表
CREATE TABLE IF NOT EXISTS students (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name TEXT NOT NULL,
    age INTEGER NOT NULL,
    major TEXT NOT NULL,
    grade TEXT,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

-- 3. 创建课程表 (含外键关联教师)
CREATE TABLE IF NOT EXISTS courses (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name VARCHAR(100) NOT NULL,
    credit INTEGER NOT NULL CHECK(credit >= 1 AND credit <= 6),
    teacher_id INTEGER NOT NULL,
    description TEXT,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (teacher_id) REFERENCES teachers(id)
);

-- 4. 创建选课关联表 (多对多关系)
CREATE TABLE IF NOT EXISTS enrollments (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    student_id INTEGER NOT NULL,
    course_id INTEGER NOT NULL,
    score DECIMAL(5,2),
    semester VARCHAR(20),
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (student_id) REFERENCES students(id) ON DELETE CASCADE,
    FOREIGN KEY (course_id) REFERENCES courses(id) ON DELETE CASCADE,
    -- 同一学生不能重复选同一门课
    UNIQUE(student_id, course_id)
);

4.2 创建索引 (CREATE INDEX)

-- ============================================
-- 索引创建脚本
-- ============================================

-- 加速按专业查询学生
CREATE INDEX IF NOT EXISTS idx_students_major
    ON students(major);

-- 加速按年级查询学生
CREATE INDEX IF NOT EXISTS idx_students_grade
    ON students(grade);

-- 加速按姓名查询学生
CREATE INDEX IF NOT EXISTS idx_students_name
    ON students(name);

-- 加速按院系查询教师
CREATE INDEX IF NOT EXISTS idx_teachers_department
    ON teachers(department);

-- 加速课程表与教师表的关联查询 (JOIN)
CREATE INDEX IF NOT EXISTS idx_courses_teacher_id
    ON courses(teacher_id);

-- 加速按课程名称搜索
CREATE INDEX IF NOT EXISTS idx_courses_name
    ON courses(name);

-- 加速选课记录按学生查询 (查看某个学生的所有选课)
CREATE INDEX IF NOT EXISTS idx_enrollments_student_id
    ON enrollments(student_id);

-- 加速选课记录按课程查询 (查看某门课的所有学生)
CREATE INDEX IF NOT EXISTS idx_enrollments_course_id
    ON enrollments(course_id);

-- 加速按学期查询选课记录
CREATE INDEX IF NOT EXISTS idx_enrollments_semester
    ON enrollments(semester);

-- 查询示例：学生选课详情 (多表 JOIN)
SELECT
    s.name AS 学生姓名,
    c.name AS 课程名称,
    t.name AS 授课教师,
    e.score AS 成绩,
    e.semester AS 学期
FROM enrollments e
JOIN students s ON e.student_id = s.id
JOIN courses c ON e.course_id = c.id
JOIN teachers t ON c.teacher_id = t.id
ORDER BY s.id, e.semester;
